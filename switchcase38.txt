#include<stdio.h>
int main(){
    int type;
    
 
    scanf("%d",&type);
    switch(type){
        case 1:
            printf("under warranty");
            break;
        case 2:
            printf("Limited Warranty");
            break;
        case 3:
        printf("out of warranty");
        break;
        
        default:
        printf("invalid");
        break;
    }
  
    return 0;
}
