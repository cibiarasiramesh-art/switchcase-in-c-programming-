#include<stdio.h>
int main(){

   int amt=0;

        amt=1200;
        break;
        case 2:
        amt=800;
        break;
        case 3:

        break;
        default:
        printf("invalid");
        break;
    }
    printf("call charges Rs%d",amt);
    return 0;
}
